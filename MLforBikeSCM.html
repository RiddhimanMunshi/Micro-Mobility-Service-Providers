# Import necessary libraries
import pandas as pd
import numpy as np
from sklearn.model_selection import train_test_split
from sklearn.ensemble import RandomForestRegressor
from sklearn.metrics import mean_squared_error, r2_score
from sklearn.preprocessing import OneHotEncoder
from sklearn.impute import SimpleImputer
import matplotlib.pyplot as plt
import seaborn as sns

# Load the dataset
data = pd.read_excel('Data.xlsm')
# Data preprocessing
# Handle missing values
imputer = SimpleImputer(strategy='median')
data[['temp', 'feeling_temp', 'humidity', 'windspeed']] = imputer.fit_transform(data[['temp', 'feeling_temp', 'humidity', 'windspeed']])
# Encode categorical variables
encoder = OneHotEncoder(drop='if_binary', sparse=False)  # Ensure encoder drops redundant binary variables
categorical_vars = ['season', 'holiday','workingday','weather']
#categorical_vars = ['season','weather']
encoded_data = pd.DataFrame(encoder.fit_transform(data[categorical_vars]), columns=encoder.get_feature_names_out(categorical_vars))
data_encoded = pd.concat([data, encoded_data], axis=1)
# target is count
features = ['temp','feeling_temp','humidity','windspeed'] + list(encoded_data.columns)
target = 'count'
# Split the data into training and testing sets
X_train, X_test, y_train, y_test = train_test_split(data_encoded[features], data_encoded[target], test_size=0.3, random_state=42)

# Initialize and train the RandomForest Regressor model
rf_model = RandomForestRegressor(n_estimators=100, random_state=42)
rf_model.fit(X_train, y_train)

# Make predictions on the testing set
y_pred = rf_model.predict(X_test)

# Evaluate the model
mse = mean_squared_error(y_test, y_pred)
r2 = r2_score(y_test, y_pred)
print("Mean Squared Error:", mse)
print("R-squared Score:", r2)

# Calculation of mean absolute percentage error
def mean_absolute_percentage_error(y_true, y_pred): 
    return np.mean(np.abs((y_true - y_pred) / y_true)) * 100

mape = mean_absolute_percentage_error(y_test, y_pred)
print("Mean Absolute Percentage Error (MAPE):", mape)

# Plot feature importance
importances = rf_model.feature_importances_
indices = np.argsort(importances)[::-1]  # Sort feature importances in descending order
feature_importance_df = pd.DataFrame({
    'Feature': X_train.columns[indices],
    'Importance': importances[indices]
})

# Select top 5 features
top5_features_df = feature_importance_df.head(5)

# Plot the top 5 features using seaborn

sns.barplot(x='Importance', y='Feature', data=top5_features_df,palette='deep')

plt.ylabel('Feature')

plt.show()

################################################ XGBoost Algorithm #############################################
import xgboost as xgb
import pandas as pd
from sklearn.metrics import mean_squared_error, r2_score
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import OneHotEncoder
from sklearn.impute import SimpleImputer
import matplotlib.pyplot as plt
import seaborn as sns

# Load the dataset
data = pd.read_excel('Data.xlsm')
# Data preprocessing
# Handle missing values
imputer = SimpleImputer(strategy='median')
data[['temp', 'feeling_temp', 'humidity', 'windspeed']] = imputer.fit_transform(data[['temp', 'feeling_temp', 'humidity', 'windspeed']])
# Encode categorical variables
encoder = OneHotEncoder(drop='if_binary', sparse=False)  # Ensure encoder drops redundant binary variables
categorical_vars = ['season', 'holiday','workingday','weather']
#categorical_vars = ['season','weather']
encoded_data = pd.DataFrame(encoder.fit_transform(data[categorical_vars]), columns=encoder.get_feature_names_out(categorical_vars))
data_encoded = pd.concat([data, encoded_data], axis=1)
# target is count
features = ['temp','feeling_temp','humidity','windspeed'] + list(encoded_data.columns)
target = 'count'
# Split the data into training and testing sets
X_train, X_test, y_train, y_test = train_test_split(data_encoded[features], data_encoded[target], test_size=0.3, random_state=42)


# Convert data into DMatrix format (optimized data structure for XGBoost)
dtrain = xgb.DMatrix(X_train, label=y_train)
dtest = xgb.DMatrix(X_test, label=y_test)

# Define parameters for XGBoost
params = {
    'objective': 'reg:squarederror',  # Use regression objective
    'eval_metric': 'rmse'  # Use root mean squared error as evaluation metric
}

# Train the XGBoost model
num_rounds = 100  # Number of boosting rounds
xgb_model = xgb.train(params, dtrain, num_rounds)

# Make predictions on the testing set
y_pred = xgb_model.predict(dtest)

# Evaluate the model
mse = mean_squared_error(y_test, y_pred)
r2 = r2_score(y_test, y_pred)
print("Mean Squared Error:", mse)
print("R-squared Score:", r2)

# Calculation of mean absolute percentage error
def mean_absolute_percentage_error(y_true, y_pred): 
    return np.mean(np.abs((y_true - y_pred) / y_true)) * 100

mape = mean_absolute_percentage_error(y_test, y_pred)
print("Mean Absolute Percentage Error (MAPE):", mape)

# Plot feature importance
# Get feature importances from the XGBoost model
importance_dict = xgb_model.get_score(importance_type='weight')
feature_importance_df = pd.DataFrame({
    'Feature': list(importance_dict.keys()),
    'Importance': list(importance_dict.values())
})

# Sort the DataFrame by Importance column in descending order and select top 5 features
top5_features_df = feature_importance_df.sort_values(by='Importance', ascending=False).head(5)

# Plot the top 5 features using seaborn
plt.figure(figsize=(5, 4))
sns.barplot(x='Importance', y='Feature', data=top5_features_df, palette='deep')
plt.xlabel('Importance')
plt.ylabel('Feature')
plt.title('Top 5 Feature Importance (XGBoost)')
plt.show()
